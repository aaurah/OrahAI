    const { messages, systemPrompt, model, onDelta, onDone, onError } = opts;

    try {
      const res = await fetch(`${this.baseUrl}/chat`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-internal-key": this.internalKey,
        },
        body: JSON.stringify({
          messages,
          system_prompt: systemPrompt,
          model: model ?? undefined,
        }),
      });

      if (!res.ok || !res.body) {
        const text = await res.text().catch(() => res.statusText);
        onError(`AI service error ${res.status}: ${text}`);
        return;
      }

      const reader = res.body.getReader();
      const decoder = new TextDecoder();
      let buf = "";

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buf += decoder.decode(value, { stream: true });
        const lines = buf.split("\n");
        buf = lines.pop() ?? "";

        for (const line of lines) {
          const trimmed = line.trim();
          if (!trimmed.startsWith("data: ")) continue;

          try {
            const evt = JSON.parse(trimmed.slice(6)) as {
              type: string;
              content?: string;
              error?: string;
            };

            if (evt.type === "delta" && evt.content) {
              onDelta(evt.content);
            } else if (evt.type === "done") {
              await onDone();
              return;
            } else if (evt.type === "error") {
              onError(evt.error ?? "Unknown error");
              return;
            }
          } catch (e) {
            logger.warn("Malformed SSE event:", e);
          }
        }
      }

      await onDone();
    } catch (err) {
      const message = err instanceof Error ? err.message : "Unknown error";
      logger.error("AI stream error:", message);
      onError(message);
    }
  }
}

function slugify(s: string) {
  return s.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/(^-|-$)/g, "").slice(0, 60) || "project";
}

async function uniqueProjectSlug(workspaceId: string, base: string) {
  let slug = base;
  let i = 1;
  while (await prisma.project.findFirst({ where: { workspaceId, slug, deletedAt: null } })) {
    slug = `${base}-${i++}`;
  }
  return slug;
}

async function assertWorkspaceMember(workspaceId: string, userId: string) {
  const m = await prisma.membership.findUnique({
    where: { userId_workspaceId: { userId, workspaceId } },
  });
  if (!m) throw createError("Workspace not found or access denied", 404);
  return m;
}

      id: projectId,

        { ownerId: userId },
        { workspace: { memberships: { some: { userId } } } },
      ],
    },
  });

async function assertProjectAccess(projectId: string, userId: string) {
  const project = await prisma.project.findFirst({
    where: {
      id: projectId,
      deletedAt: null,
      OR: [
        { ownerId: userId },
        { workspace: { memberships: { some: { userId } } } },
      ],
    },
  });
  if (!project) throw createError("Project not found", 404);
  return project;
}

function safePath(p: string) {
  if (!p || p.includes("..") || p.startsWith("/")) {
    throw createError("Invalid file path", 400);
  }
}

function mimeFromName(name: string): string {
  const ext = name.split(".").pop()?.toLowerCase() ?? "";
  const map: Record<string, string> = {
    js: "application/javascript",
    ts: "text/typescript",
    tsx: "text/typescript",
    jsx: "text/javascript",
    py: "text/x-python",
    html: "text/html",
    css: "text/css",
    json: "application/json",
    md: "text/markdown",
    sh: "text/x-shellscript",
    yaml: "text/yaml",
    yml: "text/yaml",
    go: "text/x-go",
    rs: "text/x-rust",
    java: "text/x-java",
    sql: "text/x-sql",
    xml: "text/xml",
    txt: "text/plain",
  };
  return map[ext] ?? "text/plain";
}

interface FlatFile {
  id: string;
  path: string;
  name: string;
  mimeType: string;
  size: number;
  isDir: boolean;
}

interface TreeNode extends FlatFile {
  children?: TreeNode[];
}

function buildTree(files: FlatFile[]): TreeNode[] {
  const map = new Map<string, TreeNode>();

  for (const f of files) {
    map.set(f.path, { ...f, children: f.isDir ? [] : undefined });
  }

  const roots: TreeNode[] = [];

  for (const [path, node] of map) {
    const parts = path.split("/");
    if (parts.length === 1) {
      roots.push(node);
      continue;
    }
    const parentPath = parts.slice(0, -1).join("/");
    const parent = map.get(parentPath);
    if (parent?.children) {
      parent.children.push(node);
    } else {
      roots.push(node);
    }
  }

  // Sort: directories first, then alphabetically
  const sortNodes = (nodes: TreeNode[]) => {
    nodes.sort((a, b) => {
      if (a.isDir !== b.isDir) return a.isDir ? -1 : 1;
      return a.name.localeCompare(b.name);
    });
    for (const n of nodes) {
      if (n.children) sortNodes(n.children);
    }
  };

  sortNodes(roots);
  return roots;
}

  async (req: AuthenticatedRequest, res: Response, next: NextFunction) => {

      await assertProjectAccess(req.params.projectId, req.user!.id);

      const schema = z.object({
        path: z.string().min(1).max(500),
        content: z.string().optional(),
        isDir: z.boolean().default(false),
      });

      const parsed = schema.safeParse(req.body);
      if (!parsed.success) return next(createError("Validation error", 400, parsed.error.errors));

      const { path, content, isDir } = parsed.data;
      safePath(path);

      // Check if already exists
      const existing = await prisma.file.findFirst({
        where: { projectId: req.params.projectId, path, deletedAt: null },
      });
      if (existing) return next(createError("File already exists", 409));

      // Extract name from path
      const name = path.split("/").pop() ?? path;
      const mimeType = isDir ? "inode/directory" : mimeFromName(name);

      const file = await prisma.file.create({
        data: {
          projectId: req.params.projectId,
          path,
          name,
          content: isDir ? null : (content ?? ""),
          mimeType,
          size: isDir ? 0 : Buffer.byteLength(content ?? "", "utf8"),
          isDir,
        },
      });

      res.status(201).json({ data: file, message: isDir ? "Directory created" : "File created" });
    } catch (err) {
      next(err);
    }
  });

      safePath(filePath);

      const file = await prisma.file.findFirst({
        where: { projectId: req.params.projectId, path: filePath, deletedAt: null },
      });

      if (!file) return next(createError("File not found", 404));

      if (file.isDir) {
        // Delete directory and all children
        await prisma.file.updateMany({
          where: {
            projectId: req.params.projectId,
            OR: [
              { path: filePath },
              { path: { startsWith: `${filePath}/` } },
            ],
            deletedAt: null,
          },
          data: { deletedAt: new Date() },
        });
      } else {
        await prisma.file.update({
          where: { id: file.id },
          data: { deletedAt: new Date() },
        });
      }

      res.json({ message: file.isDir ? "Directory deleted" : "File deleted" });
    } catch (err) {
      next(err);
    }
  });

    try {
      await assertProjectAccess(req.params.projectId, req.user!.id);

      const schema = z.object({
        oldPath: z.string().min(1).max(500),
        newPath: z.string().min(1).max(500),
      });

      const parsed = schema.safeParse(req.body);
      if (!parsed.success) return next(createError("Validation error", 400, parsed.error.errors));

      const { oldPath, newPath } = parsed.data;
      safePath(oldPath);
      safePath(newPath);

      const file = await prisma.file.findFirst({
        where: { projectId: req.params.projectId, path: oldPath, deletedAt: null },
      });

      if (!file) return next(createError("File not found", 404));

      // Check if target exists
      const existing = await prisma.file.findFirst({
        where: { projectId: req.params.projectId, path: newPath, deletedAt: null },
      });
      if (existing) return next(createError("Target path already exists", 409));

      const newName = newPath.split("/").pop() ?? newPath;
      const newMimeType = file.isDir ? "inode/directory" : mimeFromName(newName);

      if (file.isDir) {
        // Rename directory and all children
        const children = await prisma.file.findMany({
          where: {
            projectId: req.params.projectId,
            path: { startsWith: `${oldPath}/` },
            deletedAt: null,
          },
        });

        for (const child of children) {
          const childNewPath = child.path.replace(oldPath, newPath);
          await prisma.file.update({
            where: { id: child.id },
            data: { path: childNewPath },
          });
        }
      }

      const updated = await prisma.file.update({
        where: { id: file.id },
        data: {
          path: newPath,
          name: newName,
          mimeType: newMimeType,
        },
      });

      res.json({ data: updated, message: "File renamed" });
    } catch (err) {
      next(err);
    }
  });

// POST /api/runs/:projectId - Execute code
// ══════════════════════════════════════════════════════════════════════════════

router.post("/:projectId", requireAuth,
  async (req: AuthenticatedRequest, res: Response, next: NextFunction) => {
    try {
      const project = await assertProjectAccess(req.params.projectId, req.user!.id);

      const schema = z.object({ command: z.string().max(500).optional() });
      const parsed = schema.safeParse(req.body);
      if (!parsed.success) return next(createError("Validation error", 400, parsed.error.errors));

      const command = parsed.data.command ?? ENTRY_POINT[project.language] ?? "node index.js";

      const run = await prisma.run.create({
        data: {
          projectId: project.id,
          command,
          status: "queued",
        },
      });

      // Forward to sandbox service (fire-and-forget; sandbox POSTs status back)
      const sandboxUrl = config.sandboxUrl;
      fetch(`${sandboxUrl}/execute`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          runId: run.id,
          projectId: project.id,
          command,
          language: project.language,
        }),
      }).catch(async (err) => {
        logger.warn(`Sandbox request failed: ${err.message}`);
        // Mark failed if sandbox unreachable
        await prisma.run.update({
          where: { id: run.id },
          data: { status: "error", output: "Sandbox service unavailable" },
        }).catch((e: unknown) => logger.warn("Failed to update run status:", e));
      });

      res.status(202).json({ data: run });
    } catch (err) {
      next(err);
    }
  });

import { Router, Request, Response, NextFunction } from "express";
import bcrypt from "bcryptjs";
import jwt from "jsonwebtoken";
import { z } from "zod";
import { prisma } from "@orahai/db";
import { config } from "../config";

import { requireAuth, type AuthenticatedRequest } from "../middleware/auth";

const router = Router();

function signToken(userId: string, email: string) {
  return jwt.sign({ sub: userId, email }, config.auth.jwtSecret, {
    expiresIn: config.auth.jwtExpiresIn as jwt.SignOptions["expiresIn"],
  });
}

function safeUser(u: {
  id: string;
  email: string;
  name: string | null;
  username: string;
  avatarUrl: string | null;
  bio: string | null;
}) {
  return {
    id: u.id,
    email: u.email,
    name: u.name,
    username: u.username,
    avatarUrl: u.avatarUrl,
    bio: u.bio,
  };
}

// POST /api/auth/register
// ══════════════════════════════════════════════════════════════════════════════

router.post("/register", authRateLimiter,
  async (req: Request, res: Response, next: NextFunction) => {
    try {
      const schema = z.object({
        email: z.string().email(),
        password: z.string().min(8).max(128),
        username: z.string().min(3).max(32).regex(/^[a-zA-Z0-9_-]+$/),
        name: z.string().max(100).optional(),
      });

      const parsed = schema.safeParse(req.body);
      if (!parsed.success) {
        return next(createError("Validation error", 400, parsed.error.errors));
      }

      const existing = await prisma.user.findFirst({
        where: {
          OR: [
            { email: parsed.data.email },
            { username: parsed.data.username },
          ],
        },
      });

      if (existing) {
        const field = existing.email === parsed.data.email ? "email" : "username";
        return next(createError(`That ${field} is already in use`, 409));
      }

      const passwordHash = await bcrypt.hash(parsed.data.password, 12);

      const user = await prisma.user.create({
        data: {
          email: parsed.data.email,
          username: parsed.data.username,
          name: parsed.data.name ?? null,
          passwordHash,
        },
      });

      // Create default workspace for new user
      const workspace = await prisma.workspace.create({
        data: {
          name: `${user.username}'s Workspace`,
          slug: user.username,
          memberships: {
            create: { userId: user.id, role: "owner" },
          },
        },
      });

      const token = signToken(user.id, user.email);

      res.status(201).json({
        data: {
          user: safeUser(user),
          workspace: { id: workspace.id, name: workspace.name, slug: workspace.slug },
          token,
        },
      });
    } catch (err) {
      next(err);
    }
  });

        return next(createError("Invalid email or password", 401));

      const token = signToken(user.id, user.email);

      res.json({ data: { user: safeUser(user), token } });
    } catch (err) {
      next(err);
    }
  });

const router = Router();

function slugify(s: string) {
  return s.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/(^-|-$)/g, "").slice(0, 60) || "workspace";
}

async function uniqueSlug(base: string) {
  let slug = base;
  let i = 1;
  while (await prisma.workspace.findFirst({ where: { slug, deletedAt: null } })) {
    slug = `${base}-${i++}`;
  }
  return slug;
}

              description: true,
              avatarUrl: true,
              createdAt: true,
              updatedAt: true,
